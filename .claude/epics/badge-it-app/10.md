### Share Button Implementation
```vue
<template>
  <Button
    icon="pi pi-share-alt"
    label="Share"
    @click="toggleShareMenu"
  />
  <Menu ref="shareMenu" :model="shareOptions" :popup="true" />
</template>

<script setup>
const shareOptions = [
  {
    label: 'Twitter/X',
    icon: 'pi pi-twitter',
    command: () => shareToTwitter()
  },
  {
    label: 'LinkedIn',
    icon: 'pi pi-linkedin',
    command: () => shareToLinkedIn()
  },
  {
    label: 'Facebook',
    icon: 'pi pi-facebook',
    command: () => shareToFacebook()
  },
  { separator: true },
  {
    label: 'Copy Link',
    icon: 'pi pi-copy',
    command: () => copyLink()
  }
]

const shareToTwitter = () => {
  const url = encodeURIComponent(window.location.href)
  const text = encodeURIComponent('Create your own profile badge!')
  window.open(`https://twitter.com/intent/tweet?url=${url}&text=${text}`)
}
</script>
```
